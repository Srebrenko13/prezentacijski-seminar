import java.io.IOException;
import java.nio.file.*;
import java.nio.file.attribute.BasicFileAttributes;
import java.util.*;

public class ListTypeAnalyzer extends SimpleFileVisitor<Path> {

    private final Set<String> extensions;
    private final Map<String, Long> totalSizes = new HashMap<>();
    private final Map<Path, Long> dirSizes = new HashMap<>();
    private final Deque<Path> dirStack = new ArrayDeque<>();

    public ListTypeAnalyzer(List<String> extensions) {
        this.extensions = new HashSet<>();
        for (String ext : extensions) {
            this.extensions.add(ext.toLowerCase());
            totalSizes.put(ext.toLowerCase(), 0L);
        }
    }

    @Override
    public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) {
        dirStack.push(dir);
        dirSizes.put(dir, 0L);
        return FileVisitResult.CONTINUE;
    }

    @Override
    public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) {
        String fileName = file.getFileName().toString();
        String ext = extractExtension(fileName);

        if (ext != null && extensions.contains(ext)) {
            long size = attrs.size();

            // zbrajanje ukupnih veličina po ekstenziji
            totalSizes.merge(ext, size, Long::sum);

            // zbrajanje veličine u trenutnom direktoriju
            Path currentDir = dirStack.peek();
            dirSizes.merge(currentDir, size, Long::sum);
        }

        return FileVisitResult.CONTINUE;
    }

    @Override
    public FileVisitResult postVisitDirectory(Path dir, IOException exc) {
        long size = dirSizes.getOrDefault(dir, 0L);

        // ispis veličine direktorija (od dna prema vrhu)
        System.out.println(dir + " -> " + size + " bytes");

        dirStack.pop();
        return FileVisitResult.CONTINUE;
    }

    public Map<String, Long> getSizesByExtension() {
        return Collections.unmodifiableMap(totalSizes);
    }

    private String extractExtension(String fileName) {
        int dot = fileName.lastIndexOf('.');
        if (dot <= 0 || dot == fileName.length() - 1) return null;
        return fileName.substring(dot + 1).toLowerCase();
    }
}


List<String> ekstenzije = List.of("txt", "java", "md");
ListTypeAnalyzer analyzer = new ListTypeAnalyzer(ekstenzije);

Files.walkFileTree(Paths.get("C:/test"), analyzer);

Map<String, Long> rezultat = analyzer.getSizesByExtension();
System.out.println(rezultat);
